Function 随机延时(倍数)  ' 实现随机延时功能，根据倍数计算实际延时时间 
    If 倍数 <= 0.5 Then 倍数 = 0.5 
    Dim 基础延时 = Int(指令速度 * 倍数), 最长延时 = Int(指令速度 *(倍数 + 0.2)) 
    If 最长延时 > 1000 Then 最长延时 = 1000 
    zm.Delay 基础延时, 最长延时 
End Function
Function IIF(判断条件, 正确结果, 错误结果)  ' 简化条件判断语句
    If 判断条件 Then IIF = 正确结果 Else IIF = 错误结果  ' 根据条件返回相应结果
End Function
Function 转换数位格式(数值, 位数) 
    转换数位格式 = Right(String(位数, "0") & CStr(数值), 位数) 
End Function

' ========== 图色识别 ==========
Function 精查单色标识(单色标识)
    Dim 特征 = 指定标识特征(单色标识)
    KeepCapture
    Dim 查到结果 = zm.FindColor(特征(0), 特征(1), 特征(2), 特征(3), 特征(4), 1.0)
    If Not IsNull(查到结果) Then
        主控["颜色X"] = 查到结果["x"] : 主控["颜色Y"] = 查到结果["y"]
        ReleaseCapture : 精查单色标识 = True : Exit Function
    End If
    ReleaseCapture : 精查单色标识 = False   
End Function
Function 查询单色标识(单色标识)  ' 查询指定区域内的单色标识
    Dim 偏色集 = Array("000000", "030303", "060606")  ' 定义偏色列表
    Dim 相似度 = Array(1.0, 0.975, 0.95)  ' 定义相似度列表
    Dim 特征 = 指定标识特征(单色标识), 待查特征 = 特征(4) & "|"  ' 获取标识参数和颜色集
    KeepCapture  ' 保持屏幕截图
    For i = 0 To UBound(相似度)  ' 遍历相似度列表
        Dim 当前相似度 = 相似度(i)  ' 获取当前相似度
        For j = 0 To UBound(偏色集)  ' 遍历偏色列表
            Dim 偏色 = 偏色集(j) : 待查特征 = Replace(待查特征, "|", "-" & 偏色 & "|")  ' 获取当前偏色并生成新颜色集
            待查特征 = UTF8.Left(待查特征, Len(待查特征) - 1)  ' 移除末尾的"|"
            Dim 查到结果 = zm.FindColor(特征(0), 特征(1), 特征(2), 特征(3), 待查特征, 当前相似度)
            If Not IsNull(查到结果) Then
                主控["颜色X"] = 查到结果["x"] : 主控["颜色Y"] = 查到结果["y"]
                ReleaseCapture : 查询单色标识 = True : Exit Function 
            End If
        Next
    Next
    ReleaseCapture : 查询单色标识 = False
End Function
Function 查询指定色数(标识名称)  ' 查询指定区域内的指定颜色数量
    Dim 特征 = 指定标识特征(标识名称)
    KeepCapture
    查询指定色数 = GetColorNum(特征(0), 特征(1), 特征(2), 特征(3), 特征(4), 0.95)
    ReleaseCapture
End Function
' ========== 文字识别 ==========
Function 识别指定文字(指定文字)  ' 识别指定区域内是否存在指定文字
    Dim 特征 = 指定标识特征(指定文字), 识别结果 = OcrEx(特征(0), 特征(1), 特征(2), 特征(3)) ': TracePrint zm.VarInfo(识别结果)
    主控["文字X"] = "" : 主控["文字Y"] = "" : 主控["识别结果"] = ""
    If Len(识别结果) = 1 And 识别结果[1]["text"] = "" Then 识别指定文字 = False : Exit Function 
    For i = 1 To Len(识别结果)
        Dim 识别文本 = 识别结果[i]["text"], 指定文本 = 特征(4)
        If 识别文本 <> "" And 识别结果[i]["score"] >= 0.7 And InStr(1, 识别文本, 指定文本) > 0 Then
            主控["文字X"] = 识别结果[i]["x"] + Int(识别结果[i]["w"] / 2)
            主控["文字Y"] = 识别结果[i]["y"] + Int(识别结果[i]["h"] / 2)
            主控["识别结果"] = 识别文本
            识别指定文字 = True : Exit Function
        End If
    Next
    识别指定文字 = False
End Function
Function 识别首行文字(首行标识)  ' 识别指定区域内的首行文本信息
    Dim 特征 = 指定标识特征(首行标识), 识别结果 = OcrEx(特征(0), 特征(1), 特征(2), 特征(3)) ': TracePrint zm.VarInfo(识别结果)

    主控["文字X"] = 识别结果[1]["x"] + Int(识别结果[1]["w"] / 2)
    主控["文字Y"] = 识别结果[1]["y"] + Int(识别结果[1]["h"] / 2)

' ========== 核查文件完整性 ==========
Function 核查文件完整性 
    运行阶段("路径检测") 
    If Dir.Exist(基础路径) <> 2 Then 
        zm.DirCreate(基础路径) : 记录信息("路径缺失:自动创建路径") 
    Else 
        记录信息("文件路径:文件路径[完整]") 
    End If 
    运行阶段("配置检测") 
    If Dir.Exist(配置文件) <> 1 Then 
        记录信息 "文件缺失:挂机前先配置": 监控信息 ("文件缺失:挂机前先配置")
        随机延时(20) : 核查文件完整性 = False : Exit Function 
    Else 
        记录信息("配置文件:配置文件[完整]") 
    End If 
    运行阶段("错误记录") 
    If Dir.Exist(错误文件) <> 1 Then 
        zm.FileWrite(错误文件, "") : 记录信息("错文缺失:自动创建错文") 
    Else 
        Dim 总行数 = zm.FileLinesNumber(错误文件) 
        If 总行数 > 2000 Then 
            Dim 新内容 = zm.FileReadLine(错误文件, 总行数 - 1999, 2000) 
            zm.FileWrite(错误文件, 新内容) 
        End If 
        记录信息("记录文件:最新千行[完整]") 
    End If 
    核查文件完整性 = True 
End Function

    End If

    For 2 ' 闪烁提示       	
        运行信息 ("没有设置:明确集资统帅") : zm.Delay (1500)

        For j = 1 To Len(识别结果)
            当前弹窗 = 识别结果[j]["text"]
            If 当前弹窗 <> "" And 识别结果[j]["score"] >= 0.7  And InStr(1, 弹窗列表, 当前弹窗 & "|") > 0 Then 
                Select Case 当前弹窗 
                Case "任务完成" : 监控信息 ("检测出现:任务领奖弹窗") 
                Case "首次加入军团" : 监控信息 ("检测出现:加入军团弹窗") 
                Case Else :监控信息 ("检测出现:" & UTF8.Left(当前弹窗, 3) & "弹窗界面")
                End Select
                出现弹窗 = True : Exit For
            End If   
        Next
        If 出现弹窗 Then 
            监控阶段 ("弹窗处理")
            Select Case 当前弹窗 ' 不核查是否成功处理,不成功下轮循环再次处理
            Case "任务完成"
                清除任务弹窗  : 监控信息 ("成功处理:任务领奖弹窗")
            Case "首次加入军团"  
                zm.KeyPress 111 : 随机延时 (2) : 监控信息 ("成功退出:加入军团弹窗")
            Case Else
                For j = 1 To Len(识别结果)
                    当前按钮 = 识别结果[j]["text"]
                    If 当前按钮 <> "" And 识别结果[j]["score"] >= 0.7 And InStr(1, 按钮列表, 当前按钮 & "|") > 0 Then 
                        主控["弹窗X"] = 识别结果[j]["x"] + Int(识别结果[j]["w"] / 2)
                        主控["弹窗Y"] = 识别结果[j]["y"] + Int(识别结果[j]["h"] / 2)
                        单击指定按钮 ("弹窗退出按钮"):   监控信息 ("成功" & 当前按钮 & ":" & UTF8.Left(当前弹窗, 3) & "弹窗界面")
                    End If
                Next
            End Select
        Else 
            随机延时 (1)
        End If
    Next
End Function

Function 异常监控系统 
    Dim 卡屏计数 = 0, 初始色数, 临时色数, 循环次数 
    加载全局配置
    Do 
        If 获取共享数据("游戏状态") = "卡屏状态" Then 监控阶段("系统异常") : 监控信息("自纠状态:暂停异常监控") : Goto 下轮监控 
        监控阶段 ("卡屏监测") : 监控信息 ("正在监测:游戏运行状态") : 初始色数 = 监测卡屏色数()
        监测处理弹窗 
        监控阶段 ("卡屏验证") : 监控信息 ("正在验证:游戏运行状态") : 临时色数 = 监测卡屏色数()
        If 临时色数 <> 初始色数 Then 监控信息("验证结果:游戏运行[正常]") : Goto 下轮监控 
        初始色数 = 临时色数 : 卡屏计数 = 卡屏计数 + 1 
        If 卡屏计数 >= 尝试次数 Then 设置共享数据("游戏状态", "卡屏状态") : 卡屏计数 = 0  : zm.Delay (100) : Goto 下轮监控

        If 配置信息["挂机模式"] = "手动控制" Then zm.Delay 5000 
    Loop 
End Function

Function 返回进入战场 
    Dim 界面名称 = 主控["识别结果"] & "界面", 按钮名称 = 主控["识别结果"] & "按钮", 功能名称 = UTF8.Left(主控["识别结果"], 2)
    主控["按钮X"] = 主控["文字X"]
    For i = 1 To 尝试次数 
        If i = 1 Or i Mod 3 = 0 Then 单击指定按钮 ("返场进场按钮")
        If 识别指定文字(界面名称) Then 
            For j = 1 To 尝试次数 
                If j = 1 Or j Mod 3 = 0 Then  单击指定按钮 (按钮名称)
                If Not 识别指定文字(界面名称) Then 运行信息 ("开始" & 功能名称 & ":" & 界面名称) : 返回进入战场 = True : Exit Function
                随机延时 (j / 2)
            Next 
        End If
        随机延时 (i / 2)
    Next 
    记录信息 ("无法" & 功能名称 & ":" & 界面名称) : 返回进入战场 = False : 触发卡屏重启 
End Function
Function 重返统帅战场 
    Dim 弹窗列表 = Array("进入游戏", "重磅开启", "超值礼包", "任务活动", "指挥总部", "加入军团", "流亡状态"), 查弹窗次 = 6
    随机延时 (5) : 运行信息 ("正在重返:统帅战场界面")
    For i = 1 To 尝试次数 
        For Each 弹窗名称 In 弹窗列表 
            If 识别指定文字(弹窗名称 & "弹窗") Then 
                运行信息 ("检测出现:" & 弹窗名称 & "弹窗")
                Select Case 弹窗名称 
                Case "进入游戏" 
                    For j = 1 To 查弹窗次
                        单击指定按钮("进入游戏按钮") 
                        If Not 识别指定文字("进入游戏弹窗") Then 运行信息 ("成功处理:进入游戏弹窗") : Exit For
                        随机延时 (j / 2)
                    Next
                Case "指挥总部" ' 日后完善: 3次,找到中部但找不到 按钮 ;才判定
                    Dim 计数 = 0
                    For j = 1 To 查弹窗次
                        If 识别指定文字("指挥总部弹窗") And 识别指定文字("返场进场标识") Then 计数 = 计数 + 1
                        随机延时 (j / 2)
                    Next    
                    If 计数 >= 查弹窗次 / 2 Then 
                        运行信息 ("统帅在线:重返统帅战场") : 设置共享数据 "统帅状态", "在线状态" : 返回进入战场 
                    Else 
                        运行信息 ("统帅下线:跳过[" & 获取共享数据("目标统帅") & "]") : 设置共享数据 "统帅状态", "下线状态"
                    End If
                Case Else 
                    For j = 1 To 查弹窗次
                        zm.KeyPress 111 : 随机延时 (2 + j)
                        If Not 识别指定文字(弹窗名称 & "弹窗") Then 运行信息 ("成功退出:" & 弹窗名称 & "弹窗") : Exit For
                    Next
                End Select
            Else 
                随机延时 (1)
            End If
        Next 
        If 识别指定文字("统帅战场界面") Then 运行信息 ("成功重返:统帅战场界面") : 清除任务弹窗  : Exit Function
    Next 
    记录信息("无法重返:统帅战场界面") : 触发卡屏重启 
End Function
Function 平台卡屏重启(重启类型) 
    Dim 当前APP = 获取共享数据("当前包名"),当前平台 = 获取共享数据("当前平台")
    关闭当前平台(重启类型, 当前APP, 当前平台) 
    双清系统缓存 
    If 启动当前平台(重启类型, 当前APP, 当前平台) Then 
        运行信息("正在恢复:" & 当前平台 & "客户端手游") : 随机延时(1) 
        If 重返统帅战场() Then 运行信息 ("成功恢复:卡屏" & 当前平台)
        Exit Function 
    End If
    记录信息("原APP失败，尝试备用平台") 
    当前APP = 获取共享数据("备用包名") : 当前平台 = 获取共享数据("备用平台") ' 载入备用平台数据
    关闭当前平台(重启类型, 当前APP, 当前平台)

    运行信息("正在恢复:备用" & 当前平台) : 随机延时(1) 
    If 重返统帅战场() Then 运行信息("恢复备用:" & 当前平台 & "客户端手游") 
End Function

Function 标准战场界面 
    Dim 特征 = 指定标识特征("标准界面标识"), 点数 = UBound(特征) / 2, 颜色列表 = Split(特征(UBound(特征)), "|") 
    KeepCapture 
    For i = 0 To 点数 - 1 
        If GetPixelColor(特征(i * 2), 特征(i * 2 + 1)) <> 颜色列表(i) Then ReleaseCapture : 标准战场界面 = False : Exit Function 
    Next 
    ReleaseCapture : 标准战场界面 = True 
End Function

            单击指定按钮 (按钮名称)
        Next
        If 标准战场界面() Then 运行信息 ("成功初始:标准战场界面") : 初始战场界面 = True : Exit Function
        If i Mod 3 = 0 Then 清除任务弹窗
    Next

Function 查询统帅信息
    运行信息("正在查询:当前统帅信息")    
    主控["当前统帅"] = "未知统帅" : 主控["当前段位"] = "未知段位" : 主控["当前军衔"] = "未知军衔"
    For i = 1 To 尝试次数
        If 识别全部文字("当前统帅信息") Then 
            Dim 识别结果 = 主控["识别结果"]
            For j = 1 To Len(识别结果)
                Dim 识别文本 = 识别结果[j]["text"]
                If 识别文本 <> "" And 识别结果[j]["score"] >= 0.7 Then
                    Dim X = 识别结果[j]["x"]
                    Select Case True
                    Case X < 1100 : 主控["当前统帅"] = 识别文本
                    Case X > 1300 : 主控["当前段位"] = IIF(UTF8.Len(识别文本) = 2, 识别文本 & "Ⅰ", Replace(识别文本, "川", "Ⅲ"))
                    Case Else : 主控["当前军衔"] = 识别文本
                    End Select
                End If
            Next
            If InStr(1, "一级|二级|三级|四级|准尉|少尉|中尉|上尉|少校|中校|上校|准将|少将|中将|上将|五星|元帅", UTF8.Left(主控["当前军衔"], 2)) > 0 Then Exit For
        End If 
        If i = 尝试次数 Then 运行信息("无法查到:当前统帅信息") : Exit For
        随机延时 (i / 2) : If i Mod 3 = 0 Then 单击指定按钮 ("进入统帅按钮") : 单击指定按钮 ("统帅帐号按钮")
    Next
    设置共享数据("统帅信息", 主控["当前统帅"] & "：" & 主控["当前段位"] & "/" & 主控["当前军衔"])
    运行信息(主控["当前统帅"] & "：" & 主控["当前段位"] & "/" & 主控["当前军衔"]) : 查询统帅信息 = True
End Function

Function 查询领土信息 
    运行阶段("领土信息") : 运行信息("正在查询:全国领土信息") 
    Dim 城区 = 0, 政区 = 0, 军区 = 0, 军区总数 = 0
    For i = 1 To 尝试次数
        If 识别全部文字("全国领土信息") Then 
            Dim 识别结果 = 主控["识别结果"]
            For j = 1 To Len(识别结果)
                Dim 识别文本 = 识别结果[j]["text"]
                If 识别文本 <> "" And 识别结果[j]["score"] >= 0.7 Then
                    Dim Y = 识别结果[j]["y"] 
                    If Y < 220 Then 
                        城区 = CInt(识别文本) 
                    ElseIf Y > 230 And Y < 270 Then 
                        政区 = CInt(识别文本) 
                    ElseIf Y > 470 Then 
                        军区 = CInt(zm.RegExMatchSingle(识别文本, "(%d+)/"))
                        军区总数 = CInt(zm.RegExMatchSingle(识别文本, "/(%d+)"))
                    End If 
                End If 
            Next       
            If 城区 + 政区 + 军区 + 军区总数 > 5 Then Exit For
        End If 
        If i = 尝试次数 Then 运行信息("无法查到:全国领土信息") : Exit For
        随机延时 (i / 2) : If i Mod 3 = 0 Then 单击指定按钮 ("内政城资按钮") : 单击指定按钮 ("内政全国按钮") ' 退回-重换 : 防界面不刷新 
    Next
    主控["城区数量"] = 城区 : 主控["政区数量"] = 政区 : 主控["军区数量"] = 军区
    运行信息("城区[" & 城区 & "]/政区[" & 政区 & "]/军区[" & 军区 & "]")
End Function

Function 查询政区信息 
    运行阶段("政区信息") : 免费加速建筑 : 运行信息("正在查询:当前政区信息") 
    Dim 编号 = "", 名称 = "", 状态 = "待建中", 等级 = 0
    For i = 1 To 尝试次数
        If 识别全部文字("当前政区信息") Then 
            Dim 识别结果 = 主控["识别结果"]
            For i = 1 To Len(识别结果) 
                Dim 识别文本 = 识别结果[i]["text"] 
                If 识别文本 <> "" And 识别结果[i]["score"] >= 0.7 Then 
                    Dim 文本中点Y = 识别结果[i]["y"] + Int(识别结果[i]["h"] / 2) 
                    If 文本中点Y > 490 And 文本中点Y < 530 Then 
                        If 建单信息勘误(识别文本) Then 
                            名称 = 主控["建单名称"] 
                            编号 = zm.RegExMatchSingle(识别文本, "%d+号") 
                            If IsNull(编号) Then 编号 = "" 
                        End If 
                    ElseIf 文本中点Y > 540 And 文本中点Y < 580 Then 
                        状态 = 识别文本 
                    ElseIf 文本中点Y > 730 And 文本中点Y < 770 Then 
                        等级 = CInt(zm.RegExMatchSingle(识别文本, "(%d+)")) 
                    End If 
                End If 
            Next
            If 名称 <> "" And 等级 > 0 Then Exit For
        End If
        If i = 尝试次数 Then 运行信息("无法查到:当前政区信息") : Exit For
        随机延时 (i / 2) : If i Mod 3 = 0 Then 单击指定按钮 ("内政城资按钮") : 单击指定按钮 ("内政建设按钮") ' 切换-重查 : 防界面不刷新  
    Next
    主控["政区编号"] = 编号 : 主控["政区名称"] = 名称 : 主控["政区状态"] = 状态 : 主控["政区等级"] = 等级 : 主控["政区信息"] = 编号 & 名称
    If UTF8.Right(主控["政区政类"], 2) = "城市" Then 主控["城区状态"] = 状态 : 主控["城区等级"] = 等级 : 主控["城区信息"] = 编号 & 名称
    运行信息 (主控["政区信息"] & "/" & 主控["政区状态"] & "/" & 主控["政区等级"] & "级") : 查询政区信息 = True
End Function

Function 查询政区资源 
    IF 主控["政区政类"] <> "城市" Then Exit Function ' 择期改善:考虑刚开始时初始查询和过程中切换城区前的区别,力求最新数据 
    Dim 政区信息 = 主控["政区信息"] : 主控["主城资源"] = "资源初始"
    运行阶段("政区资源") : 运行信息("正在查询:" & 政区信息 & "[资源]") : FW.Hide("浮窗") : 随机延时(0.5) 
    Dim 军费 = 0, 钢铁 = 0, 橡胶 = 0, 石油 = 0, 库容 = 0 
    For i = 1 To 尝试次数 
        单击指定按钮 ("清除弹出消息")
        If 识别全部文字("政区资源信息") Then 
            Dim 识别结果 = 主控["识别结果"] 
            For j = 1 To Len(识别结果) 
                Dim 识别文本 = 识别结果[j]["text"] 
                If 识别文本 <> "" And 识别结果[j]["score"] >= 0.7 Then 
                    Dim X = 识别结果[j]["x"] 
                    识别文本 = CStr(Replace(识别文本, "Y", "")) 
                    识别文本 = CStr(Replace(识别文本, "M", "*10^6")) 
                    If X < 700 Then 
                        军费 = Int(zm.Calculator(识别文本))
                    ElseIf X > 1000 And X < 1500 Then 
                        钢铁 = zm.RegExMatchSingle(识别文本, "([^/]+)/") 
                        钢铁 = zm.Calculator(IIF(IsNull(钢铁), "0", 钢铁))
                    ElseIf X > 1600 And X < 2000 Then 
                        橡胶 = zm.RegExMatchSingle(识别文本, "([^/]+)/") 
                        橡胶 = zm.Calculator(IIF(IsNull(橡胶), "0", 橡胶))
                    ElseIf X > 2000 And X < 2400 Then 
                        石油 = zm.RegExMatchSingle(识别文本, "([^/]+)/") 
                        石油 = zm.Calculator(IIF(IsNull(石油), "0", 石油))
                        库容 = zm.RegExMatchSingle(识别文本, "/([^/]+)") 
                        库容 = zm.Calculator(IIF(IsNull(库容), "0", 库容))
                    End If 
                End If 
            Next     
            If 军费 + 钢铁 + 橡胶 + 石油 + 库容 > 0 Then Exit For 
            If i = 尝试次数 Then 记录信息("无法查到:" & 政区信息 & "[资源]") : Exit For
            随机延时 (i / 2) : If i Mod 3 = 0 Then 单击指定按钮 ("内政城资按钮") : 单击指定按钮 ("内政建设按钮")
        End If  
    Next 
    主控["城区军费"] = 军费 / 20 : 主控["城区钢铁"] = 钢铁 : 主控["城区橡胶"] = 橡胶 : 主控["城区石油"] = 石油 : 主控["城区库容"] = 库容 : FW.Show ("浮窗")
    If 主控["当前城类"] = "主城" Then 主控["主城军费"] = 主控["城区军费"] : 主控["主城钢铁"] = 钢铁 : 主控["主城橡胶"] = 橡胶 : 主控["主城石油"] = 石油 : 主控["主城库容"] = 库容
    运行信息 (Round(军费 / 10 ^ 6, 2) & "/" & Round(钢铁 / 10 ^ 6, 2) & "/" & Round(橡胶 / 10 ^ 6, 2) & "/" & Round(石油 / 10 ^ 6, 2) & "/" & Round(库容 / 10 ^ 6, 2) & "[M]")
End Function

Function 分类政区建单(建单编号, 建单名称, 建单状态, 建单等级)

    主控["汇总建单"] = 主控["汇总建单"] & "|" & 当前建单 ' 判断是否重复的依据 
    标记塔防在建状态(建单名称, 建单状态) ' 主城有塔防正在建设(升级中/建造中/重建中),标识供主线跳过升级塔防
    统计已建内塔 建单名称, 建单等级
    统计满级高建 建单名称, 建单状态, 建单等级
    If 统计重建建单(当前建单, 建单名称, 建单状态) Then Exit Function '重建建单不限级 
    Dim 建单限级 = zm.RegExMatchSingle(主控["建单设置"], 建单名称 & "/(%d+)")

        End If 
    Next 
    运行信息("成功整理:" & 政区信息 & "[建单]") 
End Function

Function 清除建单竖线  ' 清除建筑清单中多余的竖线分隔符 
    Dim 变量数组 = Array("可建前资", "可建后资", "可建建造", "可建重建", "可建军事", "可建塔防", "可建后勤") 
    For i = 0 To UBound(变量数组)

        If Len(主控[变量数组(i)]) > 0 And UTF8.Right(主控[变量数组(i)], 1) = "|" Then 主控[变量数组(i)] = UTF8.Left(主控[变量数组(i)], Len(主控[变量数组(i)]) - 1) ' 清除末尾多余竖线

End Function 
Function 平衡资源建设 
    If 配置信息["资建排序"] <> "自动平衡" Then 平衡资源建设 = 配置信息["资建排序"] : Exit Function 
    Dim 资源建筑, 资源数量, 临时建筑, 临时数量 
    If 主控["政区政类"] = "城市" Then 
        Select Case 主控["主城等级"] 
        Case 1, 2, 3, 4 
            平衡资源建设 = "铁矿/农场" : Exit Function 
        Case 5, 6, 7, 8, 9 
            资源建筑 = Array("农场", "铁矿") : 资源数量 = Array(主控["城区军费"], 主控["城区钢铁"]) 
        Case 10, 11, 12, 13, 14 
            资源建筑 = Array("农场", "铁矿", "橡胶厂") : 资源数量 = Array(主控["城区军费"], 主控["城区钢铁"], 主控["城区橡胶"]) 
        Case Else 
            资源建筑 = Array("农场", "铁矿", "橡胶厂", "油井") : 资源数量 = Array(主控["城区军费"], 主控["城区钢铁"], 主控["城区橡胶"], 主控["城区石油"]) 
        End Select
    Else 
        资源建筑 = Array("铁矿", "橡胶厂", "油井") : 资源数量 = Array(主控["城区钢铁"], 主控["城区橡胶"], 主控["城区石油"])
    End If   
    For i = 0 To UBound(资源建筑) - 1' 冒泡排序 
        For j = i + 1 To UBound(资源建筑) 
            If 资源数量(i) > 资源数量(j) Then 
                临时数量 = 资源数量(i) : 资源数量(i) = 资源数量(j) : 资源数量(j) = 临时数量 
                临时建筑 = 资源建筑(i) : 资源建筑(i) = 资源建筑(j) : 资源建筑(j) = 临时建筑 
            End If 
        Next 
    Next 
    Dim 排序结果 = 资源建筑(0) 
    For i = 1 To UBOUND(资源建筑) : 排序结果 = 排序结果 & "/" & 资源建筑(i) : Next
    平衡资源建设 = 排序结果 
End Function

Function 平衡资源策略
    Dim 策略设置 = 配置信息["速产策略"], 资源策略, 资源数量, 最低策略, 最低数量

    End If
    Select Case 主控["主城等级"]

        平衡资源策略 = "禁止施策" : 运行信息("主城限级:禁止施放策略") : Exit Function

Function 查询主线信息
    For i = 1 To 尝试次数 
        If 识别首行文字("主线任务信息") And InStr(1, "建造|城市|拥有|生产|组建", UTF8.Left(主控["识别结果"], 2)) > 0 Then Exit For
        If i = 尝试次数 Then 运行信息("没有查到:主线任务信息") : 查询主线信息 = False : Exit Function
        随机延时 (i / 2) : If i Mod 3 = 0 Then 清除任务弹窗
    Next
    Dim 文本 = 主控["识别结果"] : 任务信息勘误(文本) ' 获取 主控["任务名称"] + 主控["任务清单"] + 主控["任务厂级"] 数据
    Select Case  UTF8.Left(文本, 2)
    Case "建造"
        主控["任务功能"] = "建造" : 主控["任务界面"] = "建造建筑" : 主控["主线信息"] = "建造" & 主控["任务名称"]
    Case "拥有", "城市"
        主控["任务等级"] = CInt(zm.RegExMatchSingle(文本, "(%d+)级"))  
        主控["任务功能"] = "升级" :  If 主控["任务等级"] = 1 Then 主控["任务功能"] = "建造"
        主控["任务界面"] = 主控["任务功能"] & "建筑" : 主控["主线信息"] = "升级" & 主控["任务等级"] & "级" & 主控["任务名称"]
    Case "生产"
        主控["任务数量"] = CInt(zm.RegExMatchSingle(文本, IIF(InStr(1, 文本, "个") > 0, "(%d+)个", "(%d+)")))
        主控["任务功能"] = IIF(InStr(1, 文本, "个") > 0, "生产军备", "生产补给")
        主控["任务界面"] = IIF(InStr(1, 文本, "个") > 0, "军备队列", "补给队列")
        主控["工厂名称"] = IIF(InStr(1, 文本, "个") > 0, "兵工", "补给")
        主控["主线信息"] = IIF(InStr(1, 文本, "个") > 0, "生产" & 主控["任务数量"] & "个" & 主控["任务名称"], "生产" & 主控["任务数量"] & 主控["任务名称"])
    Case "组建"  
        主控["任务数量"] = CInt(zm.RegExMatchSingle(文本, "(%d+)个"))
        主控["任务功能"] = "组建部队" : 主控["任务界面"] = "组建部队" : 主控["工厂名称"] = "陆军"
        主控["主线信息"] = "组建" & 主控["任务数量"] & "个" & 主控["任务名称"]
    End Select
    运行信息 ("主线信息:" & 主控["主线信息"]) : 查询主线信息 = True
End Function
Function 核查主线执行条件

    Case "建造", "升级"
        If 主控["综合建位"] < 1 Then 运行信息("建位限制:" & 主控["主线信息"]) : 核查主线执行条件 = False : Exit Function

        Dim 主线建单
        Select Case 主控["任务功能"]
        Case "建造"
            主线建单 = zm.RegExMatchSingle(主控["可建建造"], "|[^|]*" & 主控["任务名称"] & "/可建造[^|]*")
            If IsNull(主线建单) Then 运行信息("资缺限制" & 主控["主线信息"]) : 核查主线执行条件 = False : Exit Function

            If 主控["主城高建"] < 4 Then 运行信息("高建限制:终止自动主线") : 核查主线执行条件 = False : Exit Function

                主控["可建塔防"] = Replace(主控["可建塔防"], 主线建单, "") : 核查主线执行条件 = True

Function 查询生产清单

        随机延时 (0.5) : If i Mod 3 = 0 Then 初始参数 = 特征  ' 每两次重置一次参数
        Dim 识别结果 = OcrEx(初始参数(0), 初始参数(1), 初始参数(2), 初始参数(3))
        For j = 1 To Len(识别结果)
            Dim  识别文本 = 识别结果[j]["text"]
            If 识别文本 <> "" And 识别结果[j]["score"] >= 0.7 And InStr(1, 识别文本, 待查清单) > 0 Then  ' 查到清单 = 待查清单

            End If
        Next
        If 任务功能 = "生产补给" Then Exit For '补给清单只有一行,不做第二次查询
        初始参数(1) = 初始参数(1) + 315 : 初始参数(3) = 初始参数(3) + 315 ' 军备清单再查第二行
    Next
    运行信息("无法查到:" & 任务名称 & "[清单]") : 退到战场界面("全屏") : 查询生产清单 = False
End Function

Function 切换建设政区列表 
    Dim 换前政名 = 主控["政区信息"], 换前城类 = 主控["当前城类"]
    运行信息("退出" & 换前城类  & ":" & 换前政名) 
    For i = 1 To 尝试次数 
        单击指定按钮("切换政表按钮")
        查询政区信息 
        Dim 换后政名 = 主控["政区信息"] 
        If 换后政名 <> 换前政名 Then 
            Dim 换后城类 = 主控["当前城类"], 换后政类 = 主控["政区政类"], 换后组合 = 换后城类 & 换后政类
            If 换后组合 = "主城城市" Then// 城类转换与重置
                主控["当前城类"] = "附城" : 主控["城区编号"] = 0 : 换后城类 = "附城"
            ElseIf 换后组合 = "主城军团城市" or 换后组合 = "附城军团城市" Then
                主控["当前城类"] = "军城" : 主控["城区编号"] = 0 : 换后城类 = "军城"
            End If
            If (换后城类 = "附城" and 换后政类 = "城市") or 换后城类 = "军城" Then 主控["城区编号"] = 主控["城区编号"] + 1 // 编号递增
            Dim 更新信息 = IIF(主控["当前城类"] = "主城", "换至" & 主控["城区编号"]& 主控["当前城类"] & ":" , "换至" & 主控["城区编号"] & "号" & 主控["当前城类"] & ":" )
            运行信息(更新信息 & 换后政名) : 切换建设政区列表 = True : Exit Function
        End If
        随机延时 (i / 2) : If i Mod 3 = 0 Then 单击指定按钮 ("内政城资按钮") : 单击指定按钮 ("内政建设按钮")
    Next
    记录信息("无法切换"  & 主控["当前城类"]  & ":" & 换前政名) : 触发卡屏重启
End Function

Function 清除超三建单 ' 保留不超3级的信息

        If 主控[信息名称] <> "" Then 
            Dim 旧条目 = Split(主控[信息名称], "|"), 新条目 = ""
            For i = 0 To UBound(旧条目)
                If CInt(zm.RegExMatchSingle(旧条目(i), "/(%d+)")) < 3 Then 新条目 = 新条目 & "|" & 旧条目(i)
            Next
        End If
        主控[信息名称] = 新条目
    Next
End Function

Function 自动政区建设(可建建单) 
    Dim 政区信息 = 主控["政区信息"]
    Do 
        TracePrint "可建建单::" & 可建建单
        运行阶段 ("政区建设") : 运行信息 "开始建设:" & 政区信息
        可建建单 = Split(Replace(可建建单, "|", "/"), "/") 
        For i = 1 To UBound(可建建单) Step 4 
            Dim 建单编号 = 可建建单(i), 建单名称 = 可建建单(i + 1), 建单信息 = 建单编号 & 建单名称 
            Dim 建单状态 = UTF8.Right(可建建单(i + 2), 2), 建单等级 = CInt(可建建单(i + 3))
            If 建单名称 = "底置建单" Then 底置建筑清单 (政区信息) : Goto 下一建单
            If 建单名称 = "顶置建单" Then 顶置建筑清单 (政区信息) : Goto 下一建单
            If Not 核查建单可建区域(政区信息) Then Exit Do
            If Not 查询可建建单(建单信息) Then Goto 下一建单
            If Not 选定可建清单(建单信息) Then Goto 下一建单
            If Not 查询内政建设功能(建单状态) Then Goto 下一建单
            If Not 内政进入界面(建单状态) Then Goto 下一建单
            Select Case 建单状态 
            Case "建造" 
                If Not 查询建造位置(建单信息) Then Goto 下一建单
                If Not 选定建造位置(建单信息) Then Goto 下一建单
            Case "重建", "安抚" 
                If 安抚重建缺资状态(建单状态, 建单信息) Then Exit Do
            End Select
            If Not 完成内政建设建筑(建单状态, 建单信息) Then Goto 下一建单
            更新城区高建数据 建单名称, 建单等级
            If Not 核查建单可建区域(政区信息) Then Exit Do
            If Not 查询政区建位() Then Exit Do
            Rem 下一建单 
        Next
        查询政区信息  : 可建建单 = 更新政区建单信息(政区信息) ' 更新政区建单 之前,应先查询资源

Function 主控程序
    If Not 核查分辨率DPI() Then Exit Function
    信息线程 = Thread.Start(信息显示系统)
    If Not 核查文件完整性() Then Exit Function
    初始主控数据 
    加载全局配置 
    核查集资配置 
    监控线程 = Thread.Start(异常监控系统)
    主控线程 = Thread.Start(游戏主控系统)
    Do
        If 获取共享数据("游戏状态") = "卡屏状态" Then
            Thread.Stop(主控线程)
            平台卡屏重启("卡屏重启") : 运行信息("卡屏重启:游戏主控系统")
            主控线程 = Thread.Start(游戏主控系统)
            设置共享数据("游戏状态", "正常状态")
        End If
        zm.Delay (80)
    Loop
End Function
